def meter_to_zhang_chi_cun(m):
    """
    将米转换为丈、尺、寸
    参数:
        m (float): 以米为单位的长度值
    返回:
        str: 格式化的字符串结果（如"1丈2尺3.4寸"）
    """
    if m < 0:
        return "-" + meter_to_zhang_chi_cun(-m)

    # 将米转换为寸（1米=30寸），四舍五入保留一位小数
    total_cun = round(m * 30, 1)

    # 计算丈（1丈=100寸）
    zhang = total_cun // 100
    remaining_cun_after_zhang = total_cun % 100

    # 计算尺（1尺=10寸）
    chi = remaining_cun_after_zhang // 10
    cun = remaining_cun_after_zhang % 10

    # 将丈和尺转换为整数
    zhang = int(zhang)
    chi = int(chi)

    # 处理寸的显示格式（整数去小数位，小数保留一位）
    if cun.is_integer():
        cun = int(cun)
    else:
        cun = round(cun, 1)

    return f"{zhang}丈{chi}尺{cun}寸"


# 测试示例
if __name__ == "__main__":
    test_cases = 15
    result = meter_to_zhang_chi_cun(test_cases)
    print(f"{test_cases}米 = {result}")
